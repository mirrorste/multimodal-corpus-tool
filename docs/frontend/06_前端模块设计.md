## 四、API 调用层

### 4.1 Axios 实例配置

```typescript
// api/client.ts
import axios from 'axios';

const client = axios.create({
  baseURL: process.env.REACT_APP_API_URL || '/api/v1',
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// 请求拦截器
client.interceptors.request.use(
  (config) => {
    // 添加 Token
    const token = localStorage.getItem('token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error)
);

// 响应拦截器
client.interceptors.response.use(
  (response) => response.data,
  (error) => {
    // 统一错误处理
    const message = error.response?.data?.message || '请求失败';
    console.error('API Error:', message);
    return Promise.reject(error);
  }
);

export default client;
```

### 4.2 API 调用示例

```typescript
// api/videos.ts
import client from './client';

export const videoApi = {
  // 创建视频任务
  create: (data: CreateVideoDto) =>
    client.post('/videos', data),

  // 获取视频列表
  list: (params?: ListVideosParams) =>
    client.get('/videos', { params }),

  // 获取视频详情
  get: (id: string) =>
    client.get(`/videos/${id}`),

  // 删除视频
  delete: (id: string) =>
    client.delete(`/videos/${id}`),

  // 开始处理
  process: (id: string, steps?: string[]) =>
    client.post(`/tasks/${id}/process`, { steps }),
};
```

### 4.3 Hooks 封装

```typescript
// hooks/useVideos.ts
import { useState, useEffect, useCallback } from 'react';
import { videoApi } from '../api/videos';
import type { Video, CreateVideoDto } from '../types/video';

export function useVideos(initialParams?: ListVideosParams) {
  const [videos, setVideos] = useState<Video[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [params, setParams] = useState(initialParams);

  const fetchVideos = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const result = await videoApi.list(params);
      setVideos(result.data.items);
    } catch (err) {
      setError('获取视频列表失败');
    } finally {
      setLoading(false);
    }
  }, [params]);

  useEffect(() => {
    fetchVideos();
  }, [fetchVideos]);

  const createVideo = async (data: CreateVideoDto) => {
    await videoApi.create(data);
    fetchVideos();
  };

  const deleteVideo = async (id: string) => {
    await videoApi.delete(id);
    fetchVideos();
  };

  return {
    videos,
    loading,
    error,
    params,
    setParams,
    createVideo,
    deleteVideo,
    refresh: fetchVideos,
  };
}
```

## 六、状态管理

### 6.1 Context 模式

```typescript
// context/AppContext.tsx
import React, { createContext, useContext, useReducer } from 'react';

interface AppState {
  user: User | null;
  videos: Video[];
  currentVideo: Video | null;
  isLoading: boolean;
  error: string | null;
}

type Action =
  | { type: 'SET_USER'; payload: User | null }
  | { type: 'SET_VIDEOS'; payload: Video[] }
  | { type: 'ADD_VIDEO'; payload: Video }
  | { type: 'REMOVE_VIDEO'; payload: string }
  | { type: 'SET_LOADING'; payload: boolean }
  | { type: 'SET_ERROR'; payload: string | null };

const initialState: AppState = {
  user: null,
  videos: [],
  currentVideo: null,
  isLoading: false,
  error: null,
};

function appReducer(state: AppState, action: Action): AppState {
  switch (action.type) {
    case 'SET_VIDEOS':
      return { ...state, videos: action.payload };
    // ... 其他 case
    default:
      return state;
  }
}

const AppContext = createContext<{
  state: AppState;
  dispatch: React.Dispatch<Action>;
} | null>(null);

export function AppProvider({ children }: { children: React.ReactNode }) {
  const [state, dispatch] = useReducer(appReducer, initialState);

  return (
    <AppContext.Provider value={{ state, dispatch }}>
      {children}
    </AppContext.Provider>
  );
}

export function useAppContext() {
  const context = useContext(AppContext);
  if (!context) {
    throw new Error('useAppContext must be used within AppProvider');
  }
  return context;
}
```
